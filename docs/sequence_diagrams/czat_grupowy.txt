@startuml sequence_10_group_chat_message


actor "Członek" as User
participant ":GroupChatPage" as UI <<Presentation>>
participant ":AuthMiddleware" as Auth <<Infrastructure>>
participant ":ChatService" as ChatSvc <<Business Logic>>
participant ":ChatRepository" as ChatRepo <<Data Access>>
participant ":MembershipRepository" as MemRepo <<Data Access>>
participant "m:Message" as MsgEntity <<Domain>>
participant ":MessageRepository" as MsgRepo <<Data Access>>
database ":MySQL" as DB <<Database>>

User -> UI: otwiera czat grupowy sąsiedztwa
activate UI

UI -> Auth: authenticate(token)
activate Auth
Auth --> UI: userId
deactivate Auth

UI -> ChatSvc: getGroupChat(neighborhoodId)
activate ChatSvc

ChatSvc -> ChatRepo: findByNeighborhoodId(neighborhoodId)
activate ChatRepo
ChatRepo -> DB: SELECT * FROM chats\nWHERE neighborhood_id = ?\nAND type = 'GROUP'
activate DB
DB --> ChatRepo: chat_data
deactivate DB
ChatRepo --> ChatSvc: chat
deactivate ChatRepo

ChatSvc -> MemRepo: findByUserId(userId)
activate MemRepo
MemRepo -> DB: SELECT * FROM memberships\nWHERE user_id = ?\nAND neighborhood_id = ?
activate DB
DB --> MemRepo: membership_data
deactivate DB
MemRepo --> ChatSvc: membership (status ACTIVE)
deactivate MemRepo

ChatSvc --> UI: chat
deactivate ChatSvc

UI --> User: wyświetla czat grupowy

User -> UI: wpisuje wiadomość i klika "Wyślij"

UI -> ChatSvc: sendMessage(chatId, userId, content)
activate ChatSvc

create MsgEntity
ChatSvc ->> MsgEntity: <<create>>(chatId, userId, content)

ChatSvc -> MsgRepo: save(message)
activate MsgRepo
MsgRepo -> DB: INSERT INTO messages (...)
activate DB
DB --> MsgRepo: message_id
deactivate DB
MsgRepo --> ChatSvc: savedMessage
deactivate MsgRepo

ChatSvc -> MemRepo: findByNeighborhoodId(neighborhoodId)
activate MemRepo
MemRepo -> DB: SELECT * FROM memberships\nWHERE neighborhood_id = ?
activate DB
DB --> MemRepo: all_members
deactivate DB
MemRepo --> ChatSvc: membersList
deactivate MemRepo


ChatSvc --> UI: message
deactivate ChatSvc

UI --> User: wyświetla wiadomość w czacie
deactivate UI

@enduml
