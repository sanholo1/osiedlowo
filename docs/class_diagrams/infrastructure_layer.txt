package "Infrastructure Layer" {
  
  class JwtService {
    <<Utility>>
    - secret: String
    - expirationTime: Long
    + generateToken(user: User): String
    + generateRefreshToken(user: User): String
    + validateToken(token: String): boolean
    + extractUserId(token: String): Long
    + extractClaims(token: String): Map
  }
  
  interface IPasswordHasher {
    <<Strategy>>
    + hash(password: String): String
    + verify(password: String, hash: String): boolean
  }
  
  class BcryptPasswordHasher {
    <<Strategy Implementation>>
    - saltRounds: int
    + hash(password: String): String
    + verify(password: String, hash: String): boolean
  }
  
  class AuthMiddleware {
    <<Middleware>>
    - jwtService: JwtService
    - userRepository: UserRepository
    + authenticate(request: Request): boolean
    + authorize(request: Request, roles: List<String>): boolean
    + extractUser(request: Request): User
  }
  
  class ValidationService {
    <<Utility>>
    + validateEmail(email: String): boolean
    + validatePassword(password: String): boolean
    + validateRequired(fields: Map): boolean
    + validateLength(value: String, min: int, max: int): boolean
  }
  
  class DatabaseConnection {
    <<Singleton>>
    - {static} instance: DatabaseConnection
    - connection: Connection
    - config: DatabaseConfig
    - DatabaseConnection()
    + {static} getInstance(): DatabaseConnection
    + getConnection(): Connection
    + closeConnection(): void
    + executeQuery(query: String): ResultSet
  }
  
  class ErrorHandler {
    <<Utility>>
    + handleError(error: Error): Response
    + logError(error: Error): void
    + createErrorResponse(message: String, code: int): Response
  }
  
  class Logger {
    <<Utility>>
    + info(message: String): void
    + warn(message: String): void
    + error(message: String, error: Error): void
    + debug(message: String): void
  }
  
  IPasswordHasher <|.. BcryptPasswordHasher
  
  AuthMiddleware --> JwtService : używa
  AuthMiddleware --> UserRepository : używa
  
  note right of DatabaseConnection
    Wzorzec Singleton
    Zapewnia jedną instancję
    połączenia z bazą danych
  end note
  
  note left of IPasswordHasher
    Wzorzec Strategy
    Umożliwia wymianę
    algorytmu hashowania
  end note
}
